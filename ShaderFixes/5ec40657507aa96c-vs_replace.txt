void main( 
  float4 v0 : position0,
  float4 v1 : color1,
  float4 v2 : normal0,
  float4 v3 : texcoord2,
  float2 v4 : texcoord1,
  out float4 o0 : TEXCOORD0,
  out float3 o1 : TEXCOORD1,
  out float4 o2 : SV_Position0)
{
  float4 r0;
  uint4 bitmask, uiDest;
  float4 fDest;

  o0.xyz = v1.xyz;
  o0.w = 1;
  r0.x = v0.z;
  r0.y = 1;
  r0.z = dot(r0.xy, InvProjectionMatrix._m22_m32);
  r0.x = dot(r0.xy, InvProjectionMatrix._m23_m33);
  o1.z = -r0.z / r0.x;
  o1.xy = v3.xy;
  o2.xyz = v0.xyz;
  o2.w = 1;

// This one is interesting - the output position doesn't have depth, but we
// recognise the pattern above using InvProjectionMatrix._m22_m32_m23_m33 is
// used in the unprojection pattern to get the linear depth, so we try using it
// here, which works perfectly:

float4 s = StereoParams.Load(0);
o2.x += s.x * (o1.z - s.y) / o1.z;

  return;
}
