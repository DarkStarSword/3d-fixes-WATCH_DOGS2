#include "viewport.hlsl"

void main( 
  float2 v0 : position0,
  out float4 o0 : TEXCOORD0,
  out float4 o1 : SV_Position0)
{
  float4 r0,r1;
  uint4 bitmask, uiDest;
  float4 fDest;

  o0.xyzw = float4(1,1,1,1);
  r0.xy = v0.xy;
  r0.zw = float2(1,1);
  r1.x = dot(r0.xyww, Transform._m00_m10_m20_m30);
  r1.y = dot(r0.xyww, Transform._m01_m11_m21_m31);
  r0.z = dot(r0.xyzw, Transform._m03_m13_m23_m33);
  o1.z = dot(r0.xyww, Transform._m02_m12_m22_m32);
  r0.xy = r1.xy * float2(2,2) + -r0.zz;
  o1.w = r0.z;
  o1.xy = float2(1,-1) * r0.xy;

do_hud_kill(o1);

if (IniParams[3].y) {
	// Spider crosshair was active in the previous frame, we need to adjust
	// to crosshair depth to match the weapon heat guage:
	o1.x += adjust_from_depth_buffer(0, 0);
} else {
	to_hud_depth(o1);
}

  return;
}
